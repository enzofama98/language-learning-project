Dato il seguente schema della tabella anagrafica_esercizi dell'applicazione che ti ho fornito nella knowledge base

create table public.anagrafica_esercizi (

  id uuid not null default gen_random_uuid (),

  language_code character varying(20) not null,

  lezione integer not null,

  text text not null,

  tipo_esercizio character varying(50) not null,

  descrizione text null,

  frase text null,

  opzionali jsonb null default '{}'::jsonb,

  tipo_validazione character varying(20) null default 'exact'::character varying,

  soluzione text not null,

  validazione jsonb null default '{}'::jsonb,

  active boolean null default true,

  created_at timestamp with time zone null default now(),

  updated_at timestamp with time zone null default now(),

  livello character varying(255) null,

  constraint anagrafica_esercizi_pkey1 primary key (id),

  constraint fk_esercizi_codici foreign KEY (language_code) references anagrafica_codici (codice) on delete CASCADE

) TABLESPACE pg_default;

create index IF not exists idx_anagrafica_esercizi_active on public.anagrafica_esercizi using btree (active) TABLESPACE pg_default;

create index IF not exists idx_anagrafica_esercizi_language on public.anagrafica_esercizi using btree (language_code) TABLESPACE pg_default;

create index IF not exists idx_anagrafica_esercizi_lezione on public.anagrafica_esercizi using btree (lezione) TABLESPACE pg_default;

create index IF not exists idx_anagrafica_esercizi_order on public.anagrafica_esercizi using btree (language_code, lezione, created_at) TABLESPACE pg_default;

create trigger update_anagrafica_esercizi_updated_at BEFORE

update on anagrafica_esercizi for EACH row

execute FUNCTION update_updated_at_column ();

Che corrisponde al seguente chunk del file word

STEP 2 – Completa la frase Istruzione: Tocca la parola corretta per completare la frase. Frase: "Good ___!" Opzioni: morning • thank • excuse • go Pulsante: Controlla Soluzione: morning

Trasforma il file in allegato affinchè possa essere inserito all'interno della tabella fornita così come nell'esempio.

Gli esercizi devono essere mappati nelle seguenti tipologie di esercizi:

- completa la frase

- seleziona ciò che senti

- traduci

- seleziona le coppie

se hai qualche dubbio su come mappare un esercizio presente nel file in allegato scrivimelo.

Nota bene, per gli esercizi di tipo seleziona le coppie,è importante che venga seguita la struttura descritta nel seguente esmepio di insert SQL
